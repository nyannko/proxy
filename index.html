Setup

1.Creating a clean gh-pages branch

# https://gist.github.com/ramnathv/2227408
cd /path/to/repo-name
git symbolic-ref HEAD refs/heads/gh-pages
rm .git/index
git clean -fdx

2.Gitbook auto commit script

# Run this script under the gitbook folder
gitbook_dir=$1
github_repo=$2
cat >gbsync.sh <<EOF
gitbook build
cd \$HOME/github/${github_repo}/
git co gh-pages 
cp -r \$HOME/github/gitbooks/${gitbook_dir}/_book/* \$HOME/github/${github_repo}/
git add .
dt=\$(date '+%d/%m/%Y,%H:%M')
git commit -m \$dt
git push -u origin gh-pages
git co master
EOF

bash gbsync.sh
