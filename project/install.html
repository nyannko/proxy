2.Edit setup.py

#!/usr/bin/env python
from setuptools import setup, find_packages

with open("README.md", "r") as fh:
    long_description = fh.read()

setup(
    name="examplepackage-mei", # 给本目录的所有包起的名字，注意和PyPI上的名字区分
    version="0.0.1",
    description="An example package",
    long_description=long_description,
    license="MIT Licence",
    url="https://github.com/nyannko",
    author="nyannko",
    packages=find_packages(),
    include_package_data=True,
    platforms="any",
    install_requires=[],
    entry_points={
        'console_scripts': [
            'cmd=examplepackage.test:main'
        ]
    },
    classifiers=[
        "Programming Language :: Python :: 2",
        "License :: OSI Approved :: MIT License",
        "Operating System :: OS Independent",
    ],
)

3.Build pip package to wheel file

python setup.py bdist_wheel
# or # python setup.py sdist

4.Install pip package

pip install dist/examplepackage-0.0.1-py2-none-any.whl
pip show test package

Generate openssl server keys

# generate private key. Enter pass phrase for certs/server.key: 1234 
openssl genrsa -des3 -out certs/server.key 2048

# generate csr: exampleserver, 1234
openssl req -new -key certs/server.key -sha256 -out certs/server.csr

# generate self-signed certificate
openssl x509 -req -days 365 -in certs/server.csr -signkey certs/server.key -sha256 -out certs/server.crt

# convert crt to pem
openssl x509 -in certs/server.crt -out certs/server.pem -outform PEM
