curl -w %{http_code} -m 3 -s -o /dev/null www.google.com

curl --proxy socks5h://127.0.0.1:40000 -s -o /dev/null  -w "%{time_namelookup} %{time_connect} %{time_appconnect} %{time_pretransfer} %{time_redirect} %{time_starttransfer} %{time_total}\n" www.google.com
